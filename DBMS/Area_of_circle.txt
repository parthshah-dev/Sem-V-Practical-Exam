✅ Step 1: Create the Table
CREATE TABLE areas (
    radius NUMBER,
    area   NUMBER
);

✅ Step 2: PL/SQL Code Block
DECLARE
    v_radius NUMBER;
    v_area   NUMBER;
BEGIN
    FOR v_radius IN 5..9 LOOP
        v_area := 3.14 * v_radius * v_radius;

        INSERT INTO areas (radius, area)
        VALUES (v_radius, v_area);

        DBMS_OUTPUT.PUT_LINE('Radius: ' || v_radius || '  Area: ' || v_area);
    END LOOP;

    COMMIT;
EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Error occurred while inserting data.');
END;
/
